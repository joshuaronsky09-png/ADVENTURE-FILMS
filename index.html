// FILE UPLOAD

function handleUpload(input, zone) {

  const container = document.getElementById('uploaded-' + zone);

  if(!container) return;

  Array.from(input.files).forEach(file => {

    const url = URL.createObjectURL(file);

    const div = document.createElement('div');

    div.className = 'proj-card';

    div.style.cssText = 'aspect-ratio:4/3;';

    if(file.type.startsWith('video/')) {

      div.innerHTML = `<video src="${url}" class="proj-thumb" autoplay muted loop playsinline></video><div class="proj-info"><div class="proj-name">${file.name.split('.')[0].toUpperCase()}</div><div class="proj-type">VIDEO</div></div>`;

    } else {

      div.innerHTML = `<img src="${url}" class="proj-thumb" alt="${file.name}"><div class="proj-info"><div class="proj-name">${file.name.split('.')[0].toUpperCase()}</div><div class="proj-type">FOTOGRAFÍA</div></div>`;

    }

    container.appendChild(div);

  });

  input.value = '';

}

// SCROLL REVEAL

function triggerReveal() {

  setTimeout(() => {

    const reveals = document.querySelectorAll('.reveal:not(.in)');

    const observer = new IntersectionObserver(entries => {

      entries.forEach(e => { if(e.isIntersecting) { e.target.classList.add('in'); observer.unobserve(e.target); }});

    },{threshold:0.1});

    reveals.forEach(r => observer.observe(r));

  }, 100);

}
